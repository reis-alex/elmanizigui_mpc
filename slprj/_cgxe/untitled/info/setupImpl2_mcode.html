 57        function setupImpl(obj,~,~)
 58            % Implement tasks that need to be performed only once, 
 59            % such as pre-computed constants.
 60            
 61            import casadi.*
 62q1  = SX.sym('q1');
 63qd1 = SX.sym('qd1');
 64q2  = SX.sym('q2');
 65qd2 = SX.sym('qd2');
 66q3  = SX.sym('q3');
 67qd3 = SX.sym('qd3');
 68q4  = SX.sym('q4');
 69qd4 = SX.sym('qd4');
 70torque1 = SX.sym('torque1'); 
 71torque2 = SX.sym('torque2');
 72torque3 = SX.sym('torque3');
 73torque4 = SX.sym('torque4');
 74
 75robot_path = fullfile(pwd, 'elmanizigui2.urdf');
 76% robot = importrobot(robot_path);
 77% robot.DataFormat = 'row';
 78robotacceleration = urdf2casadi.Dynamics.symbolicForwardDynamics(robot_path,0);
 79
 80opt. N = 20;  
 81opt.dt = 0.1;
 82opt.n_controls  = 4;
 83opt.n_states    = 8;
 84opt.model.function = [[qd1;qd2;qd3;qd4]; robotacceleration([q1;q2;q3;q4],[qd1;qd2;qd3;qd4],[0 0 -10],[torque1;torque2;torque3;torque4])];
 85opt.model.states   =  [q1;q2;q3;q4;qd1;qd2;qd3;qd4];
 86opt.model.controls = [torque1;torque2;torque3;torque4];
 87opt.continuous_model.integration = 'euler';
 88
 89% Define parameters
 90for i = 1:opt.N
 91    opt.parameters.name{i} = ['Ref' int2str(i)];
 92end
 93opt.parameters.dim = repmat([opt.n_states, 1;],20,1);
 94
 95% Define costs
 96Q = blkdiag(1e10*eye(opt.n_states/2),0.1*eye(opt.n_states/2));
 97R = 0.001*eye(opt.n_controls);
 98
 99opt.costs.stage.parameters = opt.parameters.name;
100opt.costs.stage.sort_parameter.fixed = [];
101opt.costs.stage.sort_parameter.var = 1:20;
102opt.costs.stage.function = @(x,u,varargin)  (x-varargin{:})'*Q*(x-varargin{:});% ...
103%                                             + u'*R*u;
104                                       
105% control and state constraints
106xbound = 20;
107opt.constraints.states.upper  = xbound*ones(opt.n_states,1);
108opt.constraints.states.lower  = -xbound*ones(opt.n_states,1);
109opt.constraints.control.upper = 1.5*ones(4,1);
110opt.constraints.control.lower = -1.5*ones(4,1);
111
112opt.input.vector = opt.parameters.name;
113opt.solver = 'ipopt';
114[solver,args_mpc] = build_mpc(opt);
115
116            obj.casadi_solver = solver;
117            obj.x0 = args_mpc.vars;
118            obj.lbx = args_mpc.lbx;
119            obj.ubx = args_mpc.ubx;
120            obj.lbg = args_mpc.lbg;
121            obj.ubg = args_mpc.ubg;
122        end

124        function u = stepImpl(obj,x,ref) % communication betzeen setImpl and stepImpl are made uniquely through obj
125            % Implement tasks that need to be performed are each time step
126            w0 = obj.x0;
127            lbw = obj.lbx;
128            ubw = obj.ubx;
129            solver = obj.casadi_solver;
130            lbw(1:2) = x;               % this enforces X0 = x (i.e., measurement input to MPC)
131            ubw(1:2) = x;
132            sol = solver('x0', w0, 'lbx', lbw, 'ubx', ubw,...
133                        'lbg', obj.lbg, 'ubg', obj.ubg,'ref',ref);
134  
135            u = full(sol.x(3)); 
136        end
